character_event = {
	id = 5000
	desc = EVTDESC5000
	picture = GFX_evt_library

	is_triggered_only = yes # on_yearly_pulse, random_events, base_lifestyle_events

	min_age = 25
	lacks_dlc = "Way of Life"
	only_capable = yes
	prisoner = no
	in_command = no

	trigger = {
		lifestyle_traits < 1

		NOR = {
			trait = blinded
			has_character_flag = LE_ST
		}
	}

	immediate = {
		set_character_flag = LE_ST
	}

	weight_multiplier = {
		factor = 1

		mult_modifier = {
			factor = 1.33
			learning >= 10
		}

		mult_modifier = {
			factor = 1.33
			learning >= 9
		}

		mult_modifier = {
			factor = 1.33
			learning >= 8
		}

		mult_modifier = {
			factor = 0.83
			trait = content
		}

		mult_modifier = {
			factor = 0.7
			trait = cynical
		}

		mult_modifier = {
			factor = 5
			trait = ambitious
		}

		mult_modifier = {
			factor = 2.5
			trait = zealous
		}

		mult_modifier = {
			factor = 0.9
			trait = detached_priest
		}

		mult_modifier = {
			factor = 1.33
			trait = scholarly_theologian
		}

		mult_modifier = {
			factor = 2
			trait = mastermind_theologian
		}
	}

	option = {
		name = EVTOPTA5000

		piety = 20

		character_event = {
			id = 5002
			days = 10
			tooltip = EVTTOOLTIP5002
		}
	}

	option = {

		name = EVTOPTB5000

		add_character_modifier = {
			name = illumination
			duration = -1
		}

		if = {
			limit = {
				OR = {
					trait = ambitious
					trait = zealous
				}
			}

			character_event = {
				id = 5001
				months = 8
				random = 180
			}
		}
		else_if = {
			limit = {
				OR = {
					trait = content
					trait = cynical
				}
			}

			character_event = {
				id = 5001
				years = 2
				random = 1095 # 3 years
			}
		}
		else = {
			character_event = {
				id = 5001
				years = 1
				random = 730 # 2 years
			}
		}
	}
}

# Spawn Great Conquerors in Shattered Worlds
character_event = {
	id = HF.1

	is_triggered_only = yes # on_startup
	hide_window = yes

	only_playable = yes

	trigger = {
		is_save_game = no
		is_shattered_world = yes

		NOR = {
			has_alternate_start_setting = {
				setting = shattered_conquerors
				option = off
			}

			has_global_flag = conquerors_setup
		}
	}

	immediate = {
		set_global_flag = conquerors_setup

		if = {
			limit = {
				has_alternate_start_setting = {
					setting = shattered_conquerors
					option = on_few
				}
			}

			while = {
				count = 4
				shattered_conqueror_spawn_effect = yes
			}
		}
		else_if = {
			limit = {
				has_alternate_start_setting = {
					setting = shattered_conquerors
					option = on_normal
				}
			}

			while = {
				count = 8
				shattered_conqueror_spawn_effect = yes
			}
		}
		else_if = {
			limit = {
				has_alternate_start_setting = {
					setting = shattered_conquerors
					option = on_many
				}
			}

			while = {
				count = 12
				shattered_conqueror_spawn_effect = yes
			}
		}
	}
}

				}

						add_trait = vaishya
						remove_trait = brahmin
					}
				}

					remove_trait = vaishya
					add_trait = brahmin

					any_dynasty_member = {

						remove_trait = kshatriya
						remove_trait = vaishya
						add_trait = brahmin
					}
				}
				else_if = {
					limit = {
						is_ruler = no

					}
					if = {
						limit = {
							spouse = { is_feudal = yes }
						}

						add_trait = kshatriya
						remove_trait = vaishya
						remove_trait = brahmin

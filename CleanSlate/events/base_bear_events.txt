# Bear generation
character_event = {
	id = bear.7

	hide_window = yes
	is_triggered_only = yes # on_yearly_pulse

	only_playable = yes

	trigger = {
		OR = {
			is_feudal = yes
			is_patrician = yes
		}

		trigger_if = {
			limit = { ai = yes }

			has_game_rule = {
				name = silly_events
				value = unrestricted
			}
		}
		trigger_else = {
			NOT = {
				has_game_rule = {
					name = silly_events
					value = off
				}
			}
		}
	}

	immediate = {
		random_list = {
			9999 = { }

			1 = {
				mult_modifier = {
					factor = 10
					trait = lunatic
				}

				mult_modifier = {
					factor = 3
					trait = paranoid
				}

				mult_modifier = {
					factor = 0.2
					trait = trusting
				}

				random_close_relative = {
					limit = {
						practical_age < 40
						host = { character = ROOT }
						ai = yes
						block_general_event_trigger = no
						NOT = { has_character_flag = secret_bear }
					}

					set_character_flag = secret_bear
				}

				character_event = { id = bear.1 }
			}
		}
	}
}

# Troubled Sleep
character_event = {
	id = bear.1
	title = bear.1.title
	desc = bear.1.desc
	picture = GFX_evt_drunk

	is_triggered_only = yes # on_yearly_pulse; above

	only_playable = yes
	war = no # Don't want to annoy the player during a war
	only_capable = yes
	prisoner = no

	trigger = {
		OR = {
			is_feudal = yes
			is_patrician = yes
		}

		trigger_if = {
			limit = { ai = yes }

			has_game_rule = {
				name = silly_events
				value = unrestricted
			}
		}
		trigger_else = {
			NOT = {
				has_game_rule = {
					name = silly_events
					value = off
				}
			}
		}

		any_close_relative = {
			has_character_flag = secret_bear
			ai = yes
			host = { character = ROOT }
			block_general_event_trigger = no
		}

		NOT = { has_character_flag = secret_bear_chain }
		is_inaccessible_trigger = no
	}

	immediate = {
		random_close_relative = {
			limit = {
				has_character_flag = secret_bear
				ai = yes
				host = { character = ROOT }
				block_general_event_trigger = no
			}

			save_event_target_as = bear_character
		}

		set_character_flag = secret_bear_chain
	}

	option = {
		name = bear.1.a

		reverse_opinion = {
			name = opinion_annoyed
			who = event_target:bear_character
			years = 5
		}
	}

	option = {
		name = bear.1.b

		add_character_modifier = {
			name = bad_nights_sleep
			months = 6
		}
	}

	after = {
		character_event = {
			id = bear.2
			days = 30
			random = 30
		}
	}
}

# A Hairy Situation
character_event = {
	id = bear.2
	title = bear.2.title
	desc = bear.2.desc
	picture = GFX_evt_courtiers_talking

	is_triggered_only = yes

	prisoner = no
	only_capable = yes

	trigger = {
		event_target:bear_character = {
			ai = yes
			prisoner = no
			host = { character = ROOT }
			is_alive = yes
		}
	}

	fail_trigger_effect = {
		if = {
			limit = {
				event_target:bear_character = {
					prisoner = yes
				}
			}

			repeat_event = {
				id = bear.2
				days = 30
			}
		}
		else = {
			event_target:bear_character = {
				clr_character_flag = secret_bear
			}

			clr_character_flag = secret_bear_chain
		}
	}

	option = {
		name = bear.2.a
	}

	option = {
		name = bear.2.b
	}

	after = {
		character_event = {
			id = bear.3
			days = 30
			random = 30
		}
	}
}

# Tongue-Tied
character_event = {
	id = bear.3
	title = bear.3.title
	desc = bear.3.desc
	picture = GFX_evt_scandal

	is_triggered_only = yes

	prisoner = no
	only_capable = yes

	trigger = {
		event_target:bear_character = {
			is_alive = yes
			ai = yes
			prisoner = no
			host = { character = ROOT }
		}
	}

	fail_trigger_effect = {
		clr_character_flag = secret_bear_chain
	}

	immediate = {
		random_close_relative = {
			limit = {
				prisoner = no
				is_adult = yes
				host = { character = ROOT }
				NOT = { character = event_target:bear_character }
			}

			save_event_target_as = family_member
		}
	}

	option = {
		name = bear.3.a

		trigger = {
			is_married = yes

			spouse = {
				NOR = {
					character = event_target:bear_character
					character = event_target:family_member
				}
			}
		}

		reverse_opinion = {
			name = opinion_flattered
			who = spouse
			years = 5
		}
	}

	option = {
		name = bear.3.b

		trigger = {
			event_target:family_member = {
				character = yes
			}
		}

		reverse_opinion = {
			name = opinion_flattered
			who = event_target:family_member
			years = 5
		}
	}

	option = {
		name = bear.3.c

		piety = 50
	}

	after = {
		character_event = {
			id = bear.4
			days = 30
			random = 30
		}

		clear_event_target = family_member
	}
}

# Bearly Legal
long_character_event = {
	id = bear.6
	title = bear.6.title
	desc = bear.6.desc
	picture = GFX_evt_feast

	is_triggered_only = yes

	prisoner = no
	only_capable = yes

	trigger = {
		event_target:bear_character = {
			is_alive = yes
			ai = yes
			prisoner = no
			host = { character = ROOT }
		}
	}

	fail_trigger_effect = {
		clr_character_flag = secret_bear_chain
	}

	immediate = {
		event_target:bear_character = {
			culture_scope = {
				save_event_target_as = original_culture
			}

			gfx_culture_scope = {
				save_event_target_as = original_gfx_culture
			}

			set_graphical_culture = bear
			culture = bear
			clr_character_flag = secret_bear
		}
	}

	option = {
		name = bear.6.a
	}

	option = {
		name = bear.6.undo

		custom_tooltip = {
			text = UNDO_BEAR_TRANSFORMATION

			event_target:bear_character = {
				set_graphical_culture = event_target:original_culture
				culture = event_target:original_gfx_culture
			}
		}
	}

	option = {
		name = bear.6.b

		trigger = {
			NOT = { trait = hunter }
		}
	}

	option = {
		name = bear.6.d

		trigger = {
			NOR = {
				trait = cannibal
				trait = lunatic
			}
		}
	}

	option = {
		name = bear.6.e
		tooltip_info = cannibal

		trigger = { trait = cannibal }

		event_target:bear_character = {
			show_scope_change = no

			death = {
				death_reason = death_execution_eaten
				killer = ROOT
			}
		}
	}

	option = {
		name = bear.6.e
		tooltip_info = lunatic

		trigger = {
			NOT = { trait = cannibal }
			trait = lunatic
		}

		event_target:bear_character = {
			show_scope_change = no

			death = {
				death_reason = death_execution_eaten
				killer = ROOT
			}
		}
	}

	option = {
		name = bear.6.f
		tooltip_info = hunter

		trigger = { trait = hunter }

		event_target:bear_character = {
			show_scope_change = no

			death = {
				death_reason = death_execution_hunt
				killer = ROOT
			}
		}

		prestige = 100
	}

	after = {
		clr_character_flag = secret_bear_chain
	}
}

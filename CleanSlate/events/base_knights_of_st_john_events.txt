### Krak des Chevaliers ###

# Grand Master writes a letter
character_event = {
	id = 8500
	desc = EVTDESC8500
	picture = GFX_evt_council
	border = GFX_event_normal_frame_religion

	is_triggered_only = yes # on_yearly_pulse, random_events, holy_order_castle_requests

	only_independent = yes
	only_capable = yes
	prisoner = no

	trigger = {
		# The Hospitallers exist
		has_landed_title = d_knights_hospitaller
		government = order_government

		d_knights_hospitaller = {
			persistent_event_target:holy_order_religion = {
				religion = ROOT
			}
		}

		# The settlement has not been built yet
		b_krakdeschevaliers = {
			has_holder = no
		}

		# Archa
		c_archa = {
			location = { num_of_settlements < 7 }

			holder_scope = {
				religion = ROOT

				NOR = {
					character = ROOT
					has_character_flag = denied_krakdeschevaliers
					has_character_flag = no_land_to_religious_orders

					top_liege = {
						has_character_flag = no_land_to_religious_orders
					}
				}
			}
		}
	}

	option = {
		name = EVTOPTA8500

		c_archa = {
			holder_scope = {
				letter_event = {
					id = 8501
					days = 7
				}
			}
		}
	}
}

# Ask permission from the Count
letter_event = {
	id = 8501
	desc = EVTDESC8501
	border = GFX_event_letter_frame_religion

	is_triggered_only = yes

	option = {
		name = EVTOPTA8501

		piety = 100

		reverse_opinion = {
			name = opinion_granted_barony_to_holy_order
			who = FROM
			years = 100
		}

		c_archa = {
			location = {
				province_event = {
					id = 8502
					days = 1
				}
			}
		}
	}

	option = {
		name = EVTOPTB8501

		piety = -10

		reverse_opinion = {
			name = opinion_angry
			who = FROM
			years = 10
		}

		religion_head = {
			show_scope_change = no

			opinion = {
				name = opinion_disappointed
				who = ROOT
				years = 5
			}
		}

		set_character_flag = denied_krakdeschevaliers
	}

	option = {
		name = EVTOPTC8501

		piety = -25
		set_character_flag = no_land_to_religious_orders

		FROM = {
			show_scope_change = no

			opinion = {
				name = opinion_angry
				who = ROOT
				years = 20
			}

			letter_event = {
				id = 8800
				days = 1
			}
		}

		religion_head = {
			show_scope_change = no

			opinion = {
				name = opinion_disappointed
				who = ROOT
				years = 10
			}
		}
	}
}

# Castle has been erected
province_event = {
	id = 8502
	desc = EVTDESC8502
	picture = GFX_evt_castle_construction
	border = GFX_event_normal_frame_religion

	is_triggered_only = yes

	option = {
		name = EVTOPTA8502

		build_holding = {
			title = b_krakdeschevaliers
			type = castle
			holder = FROMFROM
		}

		hidden_effect = {
			owner = {
				set_character_flag = event_8502

				narrative_event = {
					id = 8503
					days = 7
				}
			}

			any_player = {
				limit = {
					NOT = { character = FROM }

					OR = {
						religion_group = christian
						war_with = FROM

						capital_scope = {
							# Means we live fairly near the province where the castle is built
							distance = {
								where = ROOT
								value < 100
							}
						}
					}
				}

				narrative_event = {
					id = 8503
					days = 7
				}
			}
		}
	}
}

# News spread about the new castle
narrative_event = {
	id = 8503
	title = EVTNAME8503
	desc = EVTDESC8503
	picture = GFX_evt_crusaders
	border = GFX_event_narrative_frame_religion

	is_triggered_only = yes

	option = {
		name = EVTOPTA8503

		trigger = { has_character_flag = event_8502 }

		clr_character_flag = event_8502
		prestige = 100
	}

	option = {
		name = {
			text = EVTOPTB8503
			trigger = { religion = d_knights_hospitaller }
		}
		name = {
			text = EVTOPTC8503
			trigger = {
				NOT = { religion_group = christian }
			}
		}
		name = {
			text = EVTOPTD8503
			trigger = {
				religion_group = christian
				NOT = { religion = d_knights_hospitaller }
			}
		}

		trigger = {
			NOT = { has_character_flag = event_8502 }
		}
	}
}

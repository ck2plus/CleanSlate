## Fourth Crusade Event
# Start event (the moment the Crusade starts)
character_event = {
	id = HF.49000

	is_triggered_only = yes # on_crusade_preparation_starts
	hide_window = yes

	has_dlc = "Holy Fury"

	trigger = {
		uses_new_crusade = yes

		has_game_rule = {
			name = fourth_crusade
			value = on
		}

		has_global_flag = first_crusade_complete
		NOT = { has_global_flag = fourth_crusade_launched }

		e_byzantium = {
			holder_scope = {
				realm_size >= 50
				religion_group = ROOT
				NOT = { religion = ROOT }

				c_constantinopolis = {
					holder_scope = {
						OR = {
							character = PREVPREV
							is_vassal_or_below_of = PREVPREV
						}
					}
				}

				any_dynasty_member = {
					age >= 12
					is_ruler = no
					mercenary = no
					holy_order = no

					NOT = {
						any_heir_title = {
							always = yes
						}
					}
				}

				OR = {
					has_global_flag = qa_fourth_crusade_event

					trigger_if = {
						limit = { realm_size < 100 }
						random < 10
					}
					trigger_else_if = {
						limit = { realm_size < 200 }
						random < 15
					}
					trigger_else_if = {
						limit = { realm_size < 300 }
						random < 20
					}
					trigger_else = {
						random < 25
					}
				}
			}
		}

		d_thrace = {
			de_jure_liege = k_thrace
		}

		e_latin_empire = {
			has_holder = no
		}

		any_playable_ruler = {
			is_merchant_republic = yes
			religion = ROOT
		}
	}

	immediate = {
		e_byzantium = {
			holder_scope = {
				save_event_target_as = fourth_crusade_byzantine_emperor

				narrative_event = {
					id = HF.49001
					days = 30
				}
			}
		}
	}
}

namespace = ZE

character_event = {
	id = ZE.10000

	is_triggered_only = yes # on_yearly_pulse
	hide_window = yes

	only_playable = yes
	has_dlc = "Conclave"

	trigger = {
		mercenary = yes

		primary_title = {
			has_title_flag = ze_dyn_merc

			band_creator = {
				OR = {
					is_feudal = yes
					is_republic = yes
				}
			}
		}
	}

	immediate = {
		if = {
			limit = {
				NOT = { has_character_flag = is_mercenary_leader }
			}

			set_character_flag = is_mercenary_leader
		}

		if = {
			limit = {
				NOR = {
					has_character_flag = mercenary_ambition_1
					has_character_flag = mercenary_ambition_2
					has_character_flag = mercenary_ambition_3
					has_character_flag = mercenary_ambition_4
					has_character_flag = mercenary_ambition_5
					has_character_flag = mercenary_ambition_6
					has_character_flag = mercenary_ambition_7
					has_character_flag = mercenary_ambition_8
					has_character_flag = mercenary_ambition_9
				}
			}

			set_character_flag = mercenary_ambition_1
		}

		# Ambition up chance
		random_list = {
			10 = {
				mult_modifier = {
					factor = 4
					has_character_flag = denied_last_request
				}

				mult_modifier = {
					factor = 1.3
					trait = cruel
				}

				mult_modifier = {
					factor = 1.2
					trait = wroth
				}

				mult_modifier = {
					factor = 1.4
					trait = deceitful
				}

				mult_modifier = {
					factor = 1.6
					trait = ambitious
				}

				mult_modifier = {
					factor = 0.6
					trait = just
				}

				mult_modifier = {
					factor = 0.6
					trait = kind
				}

				mult_modifier = {
					factor = 0.4
					trait = content
				}

				mult_modifier = {
					factor = 1.5

					primary_title = {
						band_creator = {
							ROOT = {
								relative_power = {
									who = PREV
									power >= 0.7
								}
							}
						}
					}
				}

				mult_modifier = {
					factor = 1.5

					primary_title = {
						band_creator = {
							ROOT = {
								relative_power = {
									who = PREV
									power >= 1
								}
							}
						}
					}
				}

				mult_modifier = {
					factor = 1.5

					primary_title = {
						band_creator = {
							ROOT = {
								relative_power = {
									who = PREV
									power >= 1.3
								}
							}
						}
					}
				}

				increase_mercenary_ambition_level_effect = yes
			}

			60 = { }
		}

		clr_character_flag = denied_last_request

		# Random mercenary event
		random_list = {
			10 = { ### Improve on military education
				trigger = {
					OR = {
						trait = misguided_warrior
						trait = tough_soldier
						trait = skilled_tactician
					}
				}

				trigger_switch = {
					on_trigger = trait

					skilled_tactician = {
						remove_trait = skilled_tactician
						add_trait = brilliant_strategist
					}

					tough_soldier = {
						remove_trait = tough_soldier
						add_trait = skilled_tactician
					}

					misguided_warrior = {
						remove_trait = misguided_warrior
						add_trait = tough_soldier
					}
				}
			}

			10 = { ### Remove Craven
				trigger = { trait = craven }

				remove_trait = craven
			}

			10 = { ### add brave
				trigger = {
					NOR = {
						trait = craven
						trait = brave
					}
				}

				add_trait = brave
			}

			10 = { ### add diligent
				trigger = {
					NOR = {
						trait = diligent
						trait = slothful
					}
				}

				add_trait = diligent
			}

			10 = { ### Remove kind
				trigger = { trait = kind }

				remove_trait = kind
			}

			10 = { ### Remove humble
				trigger = { trait = humble }

				remove_trait = humble
			}

			10 = { ### add lustful
				trigger = {
					NOR = {
						trait = chaste
						trait = lustful
					}
				}

				add_trait = lustful
			}

			10 = { ### add wroth
				trigger = {
					NOR = {
						trait = wroth
						trait = patient
					}
				}

				add_trait = wroth
			}

			10 = { ### add cruel
				trigger = {
					NOR = {
						trait = cruel
						trait = kind
					}
				}

				add_trait = cruel
			}

			10 = { ### become wounded
				trigger = {
					NOT = { trait = wounded }
					is_maimed_trigger = no
				}

				add_trait = wounded
			}

			10 = { ### add maimed
				trigger = { can_be_maimed_trigger = yes }

				add_maimed_trait_effect = yes
			}

			10 = { ### add adventurer
				trigger = {
					NOT = { trait = adventurer }
				}

				add_trait = adventurer
			}

			10 = { ### Ask for more money
				trigger = {
					primary_title = {
						had_title_flag = {
							flag = mercenary_unemployment
							years >= 3
						}
					}
				}

				mult_modifier = {
					factor = 1.3

					primary_title = {
						had_title_flag = {
							flag = mercenary_unemployment
							years >= 10
						}
					}
				}

				mult_modifier = {
					factor = 1.2

					OR = {
						has_character_flag = mercenary_ambition_4
						has_character_flag = mercenary_ambition_5
						has_character_flag = mercenary_ambition_6
						has_character_flag = mercenary_ambition_7
						has_character_flag = mercenary_ambition_8
						has_character_flag = mercenary_ambition_9
					}
				}

				mult_modifier = {
					factor = 1.2

					OR = {
						has_character_flag = mercenary_ambition_5
						has_character_flag = mercenary_ambition_6
						has_character_flag = mercenary_ambition_7
						has_character_flag = mercenary_ambition_8
						has_character_flag = mercenary_ambition_9
					}
				}

				mult_modifier = {
					factor = 1.2

					OR = {
						has_character_flag = mercenary_ambition_6
						has_character_flag = mercenary_ambition_7
						has_character_flag = mercenary_ambition_8
						has_character_flag = mercenary_ambition_9
					}
				}

				mult_modifier = {
					factor = 1.3

					OR = {
						has_character_flag = mercenary_ambition_7
						has_character_flag = mercenary_ambition_8
						has_character_flag = mercenary_ambition_9
					}
				}

				mult_modifier = {
					factor = 1.3

					OR = {
						has_character_flag = mercenary_ambition_8
						has_character_flag = mercenary_ambition_9
					}
				}

				# Demand more money
				primary_title = {
					band_creator = {
						letter_event = { id = ZE.10031 }
					}
				}
			}

			10 = {
				trigger = {
					OR = {
						has_character_flag = mercenary_ambition_5
						has_character_flag = mercenary_ambition_6
						has_character_flag = mercenary_ambition_7
						has_character_flag = mercenary_ambition_8
						has_character_flag = mercenary_ambition_9
					}

					primary_title = {
						mercenary_siphon_factor >= 0.05
					}
				}

				# Start to pay less
				primary_title = {
					band_creator = {
						letter_event = { id = ZE.10071 }
					}
				}
			}

			20 = {
				trigger = {
					primary_title = {
						band_creator = {
							NOT = {
								any_demesne_title = {
									mercenary = yes
									real_tier = COUNT
									has_mercenary_maintenance_modifier = maintaining_mercenary_band_7
								}
							}
						}
					}
				}

				# Demand more soldiers
				primary_title = {
					band_creator = {
						save_event_target_as = target_liege
						character_event = { id = ZE.11011 }
					}
				}
			}

			20 = {
				character_event = { id = ZE.11000 } # Ask to marry child of band creator
			}

			10 = {
				trigger = {
					OR = {
						has_character_flag = mercenary_ambition_6
						has_character_flag = mercenary_ambition_7
						has_character_flag = mercenary_ambition_8
						has_character_flag = mercenary_ambition_9
					}
				}

				mult_modifier = {
					factor = 2
					has_character_flag = mercenary_ambition_7
				}

				mult_modifier = {
					factor = 4
					has_character_flag = mercenary_ambition_8
				}

				mult_modifier = {
					factor = 8
					has_character_flag = mercenary_ambition_9
				}

				# Break Free
				primary_title = {
					band_creator = {
						letter_event = { id = ZE.10076 }
					}
				}
			}

			100 = { } # Nothing happened
		}
	}
}

## Demand more money
# Band creator gets my demand
letter_event = {
	id = ZE.10031
	desc = EVTDESC_ZE_10031

	is_triggered_only = yes

	option = {
		name = EVTOPTA_ZE_10031

		transfer_scaled_wealth = {
			to = FROM
			value = 0.1
			min = 5
		}

		hidden_effect = {
			FROM = {
				primary_title = {
					clr_title_flag = mercenary_unemployment
				}

				if = {
					limit = {
						OR = {
							has_character_flag = mercenary_ambition_6
							has_character_flag = mercenary_ambition_7
							has_character_flag = mercenary_ambition_8
							has_character_flag = mercenary_ambition_9
						}
					}

					random_list = {
						30 = { # ask for even more money
							mult_modifier = {
								factor = 1.3
								has_character_flag = mercenary_ambition_7
							}

							mult_modifier = {
								factor = 1.7
								has_character_flag = mercenary_ambition_8
							}

							mult_modifier = {
								factor = 2.5
								has_character_flag = mercenary_ambition_9
							}

							mult_modifier = {
								factor = 1.5
								trait = greedy
							}

							ROOT = {
								repeat_event = {
									id = ZE.10033
									days = 90
								}
							}
						}

						60 = { }
					}
				}
			}
		}
	}

	option = {
		name = EVTOPTB_ZE_10031

		random = {
			chance = 50
			add_trait_fully_cruel_effect = yes
		}

		FROM = {
			show_scope_change = no

			set_character_flag = denied_last_request

			opinion = {
				name = opinion_denied_request
				who = ROOT
				years = 5
			}
		}
	}
}

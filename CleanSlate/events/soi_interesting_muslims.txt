### Al-Ghazali ###
province_event = {
	id = 106000
	desc = EVTDESC106000
	picture = GFX_evt_emissary

	trigger = {
		province_id = c_tus
		religions_are_standard_trigger = yes
		NOT = { has_province_modifier = famous_muslim_timer }

		owner = {
			religion_group = muslim

			top_liege = {
				religion_group = muslim
				is_adult = yes
				in_command = no

				NOR = {
					has_character_modifier = muslim_famous_timer
					trait = on_hajj

					any_courtier_or_vassal = {
						has_character_flag = al_ghazali_flag
					}
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 1000
	}

	immediate = {
		add_province_modifier = {
			name = famous_muslim_timer
			years = 5
			hidden = yes
		}
	}

	option = {
		name = EVTOPTA106000

		owner = {
			show_scope_change = no

			top_liege = {
				show_scope_change = no

				add_character_modifier = {
					name = muslim_famous_timer
					years = 5
					hidden = yes
				}

				character_event = {
					id = 106001
					days = 3
					random = 4
					tooltip = EVTTOOLTIP106001
				}
			}
		}
	}
}

# Liege get new character: Al-Ghazali
character_event = {
	id = 106001
	desc = EVTDESC106001
	picture = GFX_evt_council

	is_triggered_only = yes

	option = {
		name = EVTOPTA106001

		create_character = {
			name = "Abu Hamid"
			dynasty = 1031100
			religion = ROOT
			culture = persian
			age = 27
			female = no

			attributes = {
				learning = 8
				diplomacy = 13
			}

			add_trait = mastermind_theologian
			add_trait = mystic
			add_trait = zealous
			add_trait = just
			add_trait = faqih
		}

		new_character = {
			set_character_flag = al_ghazali_flag
		}
	}
}

			dynasty = 1031103
			culture = persian
			religion = sunni
			age = 35
			female = no

			attributes = {
				learning = 15
			}

			add_trait = diligent
			add_trait = scholar
			add_trait = fortune_builder
			add_trait = quick
			add_trait = gregarious
		}

		new_character = {
			set_character_flag = zakariya_al_qazwini_flag

			if = {
				limit = {
					ROOT = {
						can_grant_title = title_chief_qadi
					}
				}

				give_minor_title = title_chief_qadi
			}
		}
	}
}

		}

			limit = {
				NOT = { event_target:cat_hater = { always = yes } }
			}

			random_lover = {
				limit = {
					ai = yes

					is_incapable = no

					NOR = {
						trait = kind
						has_character_modifier = pet_cat
					}
				}

			}

	option = {
		name = EVTOPTA_77851

		remove_character_modifier = pet_cat

		reverse_opinion = {
			name = opinion_fulfilled_demands
			who = event_target:cat_hater
			years = 10
		}
	}

	# Give cat to friend
	option = {
		name = EVTOPTB_77851

		trigger = {
			any_friend = {
				practical_age >= 12
				ai = yes
				prisoner = no
				is_incapable = no

				NOR = {
					character = event_target:cat_hater
					has_character_modifier = pet_cat
				}
			}
		}

		remove_character_modifier = pet_cat

		reverse_opinion = {
			who = event_target:cat_hater
			name = opinion_fulfilled_demands
			years = 10
		}

		random_friend = {
			show_scope_change = no

			limit = {
				practical_age >= 12
				ai = yes
				prisoner = no
				is_incapable = no

				NOR = {
					character = event_target:cat_hater
					has_character_modifier = pet_cat
				}
			}

			save_event_target_as = new_cat_owner

			add_character_modifier = {
				name = pet_cat
				duration = -1
			}

			character_event = {
				id = 77802
				days = 2555 # Cat dies
				random = 1000
			}

			if = {
				limit = {
					event_target:new_cat_owner = {
						OR = {
							trait = kind
							trait = charitable
							trait = patient
							trait = gregarious
							trait = affectionate
							trait = playful
							trait = master_schemer
							trait = celibate
						}
					}
				}

				character_event = {
					id = 77852 # Good new owner
					days = 100
				}
			}
			else_if = {
				limit = {
					event_target:new_cat_owner = {
						OR = {
							trait = cruel
							trait = wroth
							trait = impaler
							trait = haughty
							trait = brooding
						}
					}
				}

				character_event = {
					id = 77853 # Bad new owner
					days = 100
				}
			}
		}
	}

	option = {

					practical_age >= 12
					ai = yes

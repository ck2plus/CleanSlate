# CleanSlate: why is this event in this file?
character_event = {
	id = 44482

	is_triggered_only = yes # on_settlement_construction_completed
	hide_window = yes

	only_playable = yes

	trigger = {
		FROM = { holding_type = temple }
	}

	immediate = {
		religion_authority = {
			name = temple_built
			years = 20
		}

		if = {
			limit = { has_ambition = obj_strengthen_religion }

			change_variable = {
				which = strengthen_religion
				value = 1
			}
		}
	}
}

character_event = {
	id = HL.4

	is_triggered_only = yes # on_settlement_construction_completed
	hide_window = yes

	only_playable = yes
	has_dlc = "Horse Lords"

	trigger = {
		FROM = { holding_type = temple }

		clan_title = { clan = yes }
	}

	immediate = {
		clan_title = {
			holder_scope = {
				top_liege = {
					if = {
						limit = {
							religion = ROOT

							NOR = {
								has_feud_with = ROOT
								character = ROOT
							}
						}

						clan_title = {
							clan_reverse_opinion = {
								name = opinion_built_temple
								who = PREVPREVPREV # ROOT's clan title
								years = 20
							}
						}
					}

					any_vassal = {
						limit = {
							clan = yes
							religion = ROOT

							NOR = {
								has_feud_with = ROOT
								character = ROOT
							}
						}

						clan_title = {
							clan_reverse_opinion = {
								name = opinion_built_temple
								who = PREVPREVPREVPREV # ROOT's clan title
								years = 20
							}
						}
					}
				}
			}
		}
	}
}

# Feud started between two clans because of raid and pillaging
character_event = {
	id = HL.8

	is_triggered_only = yes # on_loot_province, random_events
	hide_window = yes

	only_playable = yes
	has_dlc = "Horse Lords"

	trigger = {
		is_nomadic = yes

		FROM = {
			owner = {
				is_nomadic = yes
				is_ruler = yes
				same_realm = ROOT
				NOT = { has_feud_with = ROOT }
			}
		}
	}

	immediate = {
		clan_title = {
			FROM = {
				owner = {
					clan_title = {
						create_feud = PREVPREVPREV
					}
				}
			}
		}
	}
}

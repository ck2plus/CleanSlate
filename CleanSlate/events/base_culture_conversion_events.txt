# Province culture changes
province_event = {
	id = 55000
	desc = EVTDESC55000
	picture = GFX_evt_culture_change

	trigger = {
		owner = {
			religion = ROOT
			culture = top_liege
			NOT = { culture = ROOT }
		}

		any_neighbor_province = {
			has_owner = yes

			owner = {
				ROOT = {
					owner = {
						culture = PREVPREV
						culture = PREVPREVPREV
					}
				}
			}

			NOR = {
				has_province_modifier = depopulated_1
				has_province_modifier = depopulated_2
				has_province_modifier = depopulated_3
			}
		}

		# No flipping back to Norse culture before 1150
		trigger_if = {
			limit = {
				owner = {
					culture = norse
				}

				OR = {
					culture = swedish
					culture = norwegian
					culture = danish
					culture = norman
				}
			}

			OR = {
				year >= 1150
				non_historical_culture_names_in_alt_start_trigger = yes
			}
		}
	}

	mean_time_to_happen = {
		years = 100

		mult_modifier = {
			factor = 1.5
			owner = { stewardship < 2 }
		}

		mult_modifier = {
			factor = 1.5
			owner = { stewardship < 3 }
		}

		mult_modifier = {
			factor = 1.5
			owner = { stewardship < 4 }
		}

		mult_modifier = {
			factor = 1.5
			owner = { stewardship < 5 }
		}

		mult_modifier = {
			factor = 0.75
			owner = { stewardship >= 9 }
		}

		mult_modifier = {
			factor = 0.75
			owner = { stewardship >= 11 }
		}

		mult_modifier = {
			factor = 0.75
			owner = { stewardship >= 13 }
		}

		mult_modifier = {
			factor = 0.75
			owner = { stewardship >= 15 }
		}

		mult_modifier = {
			factor = 0.95

			owner = {
				job_treasurer = {
					has_job_action = action_administer_realm
					stewardship < 10
				}
			}
		}

		mult_modifier = {
			factor = 0.75

			owner = {
				job_treasurer = {
					has_job_action = action_administer_realm
					stewardship >= 10
				}
			}
		}

		mult_modifier = {
			factor = 0.75

			owner = {
				OR = {
					primary_title = {
						conquest_culture = PREV
					}

					ROOT = {
						OR = {
							county = {
								conquest_culture = PREVPREV
							}

							duchy = {
								holder = PREVPREV
								conquest_culture = PREVPREV
							}

							kingdom = {
								holder = PREVPREV
								conquest_culture = PREVPREV
							}

							empire = {
								holder = PREVPREV
								conquest_culture = PREVPREV
							}
						}
					}
				}
			}
		}

		mult_modifier = {
			factor = 0.75
			has_province_flag = recent_depopulation_1
		}

		mult_modifier = {
			factor = 0.5
			has_province_flag = recent_depopulation_2
		}

		mult_modifier = {
			factor = 0.25
			has_province_flag = recent_depopulation_3
		}

		mult_modifier = {
			factor = 3

			OR = {
				has_game_rule = {
					name = culture_conversion
					value = slower_normal
				}

				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots_and_slower_normal
				}
			}
		}

		mult_modifier = {
			factor = 0.25

			OR = {
				culture = zhangzhung
				culture = sumpa
			}

			owner = {
				culture = bodpa

				OR = {
					any_owned_bloodline = {
						has_bloodline_flag = bloodline_songtsen_gampo
					}

					top_liege = {
						any_owned_bloodline = {
							has_bloodline_flag = bloodline_songtsen_gampo
						}
					}
				}
			}
		}

		mult_modifier = {
			factor = 0.6

			owner = {
				any_owned_bloodline = {
					has_bloodline_flag = cultural_saintly_bloodline

					founder = {
						culture = PREVPREV
					}
				}
			}
		}

		mult_modifier = { # Roman Renaissance
			factor = 0.5

			owner = {
				culture = roman
			}

			culture_group = latin
			has_global_flag = flag_started_roman_renaissance
		}

		mult_modifier = { # Greek culture is more impervious to Romanization
			factor = 2

			owner = {
				culture = roman
			}

			culture = greek
		}

		# Effect of wonder upgrades
		mult_modifier = {
			factor = 0.75

			owner = {
				top_liege = {
					any_realm_wonder_upgrade = {
						has_wonder_upgrade_flag = culture_conversion_faster
						is_active = yes

						wonder = {
							original_wonder_owner = {
								culture_group = PREVPREVPREV
							}
						}
					}
				}
			}
		}
	}

	option = {
		name = EVTOPTA55000

		culture = owner

		if = {
			limit = { has_province_modifier = nomad_agitation }
			remove_province_modifier = nomad_agitation
		}
	}
}

# A ruler is assimilated by the local culture
character_event = {
	id = 55010
	desc = EVTDESC55010
	picture = GFX_evt_stone_church

	is_triggered_only = yes # on_five_year_pulse, random_events, ruler_culture_conversion

	only_rulers = yes
	lacks_dlc = "Rajas of India" # Controlled by Decision: convert_to_local_culture
	min_age = 16
	only_capable = yes
	prisoner = no

	trigger = {
		NOT = { culture = capital_scope }

		trigger_if = {
			limit = { independent = no }

			top_liege = {
				ROOT = {
					capital_scope = {
						culture = PREVPREV
					}
				}
			}
		}

		# See special event 55002
		trigger_if = {
			limit = { culture = norman }

			capital_scope = {
				NOT = { culture = anglo-saxon }
			}
		}

		# See special event TOG.2003, oldgods_culture_conversion.txt
		trigger_if = {
			limit = {
				OR = {
					culture = norse
					culture = norman
				}

				capital_scope = {
					OR = {
						culture = french
						culture = breton
						culture = occitan
					}
				}

				should_evolve_cultures_trigger = yes
			}

			year >= 1150
		}

		is_nomadic = no

		primary_title = {
			NOR = {
				conquest_culture = ROOT
				culture = ROOT
			}

			# The first holder of a title does not convert
			any_previous_holder = {
				NOT = { character = ROOT }
			}
		}

		NOT = { government = chinese_imperial_government }
	}

	weight_multiplier = {
		factor = 1

		mult_modifier = {
			factor = 2

			independent = no

			top_liege = {
				ROOT = {
					capital_scope = {
						culture = PREVPREV
					}
				}
			}
		}

		mult_modifier = {
			factor = 2

			primary_title = {
				ROOT = {
					capital_scope = {
						culture = PREVPREV
					}
				}
			}
		}

		mult_modifier = {
			factor = 0.5
			culture_group = capital_scope # Same culture group: slower assimilation
		}
	}

	option = {
		name = EVTOPTA55010

		# Piss off the appropriate vassals
		any_vassal = {
			show_scope_change = no

			limit = {
				culture = ROOT
				is_adult = yes
			}

			opinion = {
				name = opinion_adopted_foreign_ways
				who = ROOT
				years = 10
			}

			character_event = { id = 55011 }
		}

		culture = capital_scope
	}

	option = {
		name = EVTOPTB55010

		ai_chance = { factor = 0 }
	}
}

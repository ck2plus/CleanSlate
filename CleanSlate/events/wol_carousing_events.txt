			}
		}
	}

		name = EVTOPTA_WoL_4101 # No

		if = {
			limit = {
				NOR = {

				}
			}

		}

			limit = {
				NOT = { trait = stressed }

		else_if = {
			limit = {
				NOT = { trait = depressed }
			}

			add_trait = depressed
		}

# A fosterling in your court is calling your court inferior to the child's father's court
character_event = {
	id = 78000
	desc = EVTDESC78000
	picture = GFX_evt_guardian

	is_triggered_only = yes # on_yearly_pulse, random_events

	only_rulers = yes
	prisoner = no
	only_capable = yes

	trigger = {
		any_ward = {
			father = {
				higher_real_tier_than = ROOT
				is_alive = yes
			}
		}
	}

	weight_multiplier = {
		factor = 1
	}

	immediate = {
		random_ward = {
			limit = {
				father = {
					higher_real_tier_than = ROOT
					is_alive = yes
				}
			}

			save_event_target_as = target_ward
		}
	}

	option = {
		name = EVTOPTA78000

		scaled_wealth = -0.1

		reverse_opinion = {
			name = opinion_impressed
			who = event_target:target_ward
			years = 5
		}
	}

	option = {
		name = EVTOPTB78000

		trigger = {
			NOR = {
				trait = ambitious
				trait = content
			}
		}

		random = {
			chance = 25
			add_trait_content_effect = yes
		}
	}

	option = {
		name = EVTOPTC78000

		trigger = {
			NOT = { trait = wroth }
		}

		reverse_opinion = {
			name = opinion_upset
			who = event_target:target_ward
			years = 5
		}
	}

	option = {
		name = EVTOPTD78000
		tooltip_info = wroth

		trigger = { trait = wroth }

		reverse_opinion = {
			name = opinion_mad_at_guardian
			who = event_target:target_ward
			years = 5
		}
	}
}

# A fosterling claims to always being bullied by other children at the court
character_event = {
	id = 78020
	desc = EVTDESC78020
	picture = GFX_evt_child_bully

	is_triggered_only = yes # on_yearly_pulse, random_events

	only_capable = yes

	trigger = {
		any_ward = {
			OR = {
				trait = shy
				trait = craven
				trait = kind
				trait = trusting
			}

			is_adult = no
			practical_age >= 6
			prisoner = no
			is_incapable = no

			ROOT = {
				any_courtier = {
					is_adult = no
					practical_age >= 6

					OR = {
						trait = wroth
						trait = cruel
						trait = deceitful
						trait = envious
					}

					is_abroad = no
					prisoner = no
					is_incapable = no
					NOT = { character = PREVPREV } # ward
				}
			}
		}
	}

	immediate = {
		random_ward = {
			limit = {
				is_adult = no
				practical_age >= 6

				OR = {
					trait = shy
					trait = craven
					trait = kind
					trait = trusting
				}

				prisoner = no
				is_incapable = no
			}

			save_event_target_as = bullied_ward
		}

		random_courtier = {
			limit = {
				is_adult = no
				practical_age >= 6

				OR = {
					trait = wroth
					trait = cruel
					trait = deceitful
					trait = envious
				}

				is_abroad = no
				prisoner = no
				is_incapable = no
				NOT = { character = event_target:bullied_ward }
			}

			save_event_target_as = child_bully
		}
	}

	option = {
		name = EVTOPTA78020

		event_target:bullied_ward = {
			show_scope_change = no

			random = {
				chance = 25
				add_trait_stressed_effect = yes
			}

			opinion = {
				name = opinion_unhappy
				who = ROOT
				years = 5
			}
		}
	}

	option = {
		name = EVTOPTB78020

		event_target:bullied_ward = {
			show_scope_change = no

			if = {
				limit = {
					NOR = {
						trait = craven
						trait = shy
					}
				}

				random = {
					chance = 25
					change_martial = 1
				}
			}
			else = {
				random = {
					chance = 25
					remove_trait_shy_effect = yes
				}

				random = {
					chance = 25
					remove_trait_craven_effect = yes
				}
			}

			add_trait_silently_stressed_effect = yes

			opinion = {
				name = opinion_upset
				who = ROOT
				years = 5
			}
		}
	}

	option = {
		name = EVTOPTC78020

		reverse_opinion = {
			name = opinion_angry
			who = event_target:child_bully
			years = 5
		}
	}
}

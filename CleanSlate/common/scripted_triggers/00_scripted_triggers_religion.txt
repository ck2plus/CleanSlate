FROMs_religion_is_playable_trigger = {
	FROM = {
		show_scope_change = no

		trigger_if = {
			limit = {
				religion_group = jewish_group
				NOT = { has_dlc = "Sons of Abraham" }
			}
			has_dlc = "Sons of Abraham"
		}
		trigger_else_if = {
			limit = {
				religion_group = muslim
				NOT = { has_dlc = "The Sword of Islam" }
			}
			has_dlc = "The Sword of Islam"
		}
		trigger_else_if = {
			limit = {
				religion_group = indian_group
				NOR = {
					religion = taoist
					has_dlc = "Rajas of India"
				}
			}
			has_dlc = "Rajas of India"
		}
		trigger_else_if = {
			limit = {
				OR = {
					religion = bon
					religion = bon_reformed
				}
				NOR = {
					has_dlc = "Jade Dragon"
					has_dlc = "Holy Fury"
					has_dlc = "The Old Gods"
				}
			}
			OR = {
				has_dlc = "Jade Dragon"
				has_dlc = "Holy Fury"
				has_dlc = "The Old Gods"
			}
		}
		trigger_else_if = {
			limit = {
				religion_group = pagan_group
				NOR = {
					religion = bon
					religion = bon_reformed
					has_dlc = "The Old Gods"
					has_dlc = "Holy Fury"
				}
			}
			OR = {
				has_dlc = "The Old Gods"
				has_dlc = "Holy Fury"
			}
		}
		trigger_else_if = {
			limit = {
				religion_group = zoroastrian_group
				NOR = {
					religion = khurmazta
					has_dlc = "The Old Gods"
				}
			}
			has_dlc = "The Old Gods"
		}
		trigger_else_if = {
			limit = {
				religion = khurmazta
				NOR = {
					has_dlc = "The Old Gods"
					has_dlc = "Jade Dragon"
				}
			}
			OR = {
				has_dlc = "The Old Gods"
				has_dlc = "Jade Dragon"
			}
		}
		trigger_else_if = {
			limit = {
				religion = taoist
				NOR = {
					has_dlc = "Rajas of India"
					has_dlc = "Jade Dragon"
				}
			}
			OR = {
				has_dlc = "Rajas of India"
				has_dlc = "Jade Dragon"
			}
		}
	}
}

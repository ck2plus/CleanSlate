# can_pay_assassins_debt
# has_assassins_prerequisites

can_pay_assassins_debt = {
	trigger_if = {
		limit = {
			is_variable_equal = {
				which = assassins_debt
				value = 50
			}
		}
		wealth >= 50
	}
	trigger_else_if = {
		limit = {
			is_variable_equal = {
				which = assassins_debt
				value = 250
			}
		}
		wealth >= 250
	}
	trigger_else_if = {
		limit = {
			is_variable_equal = {
				which = assassins_debt
				value = 300
			}
		}
		wealth >= 300
	}
	trigger_else_if = {
		limit = {
			is_variable_equal = {
				which = assassins_debt
				value = 500
			}
		}
		wealth = 500
	}
}

has_assassins_prerequisites = {
	is_adult = yes

	trigger_if = { # Recruitment decision
		limit = { FROM = { society_member_of = the_assassins } }

		OR = {
			FROM = { ai = no }
			hidden_trigger = { ai = no }
			religion_group = muslim
		}
	}

	trigger_else = { # Showing interest via interface
		OR = {
			true_religion = shiite
			religion_group = muslim

			trigger_if = {
				limit = { ai = yes }

				NOT = { trait = zealous }

				any_playable_ruler = {
					religion = shiite
					distance_from_realm = { who = ROOT value < 300 }
				}
			}
		}
	}

	trigger_if = {
		limit = { is_female = yes }

		has_game_rule = {
			name = gender
			value = all
		}
	}
}

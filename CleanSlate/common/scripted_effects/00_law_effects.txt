# Council Power
decreased_council_power_effect = {
	holder_scope = {
		any_courtier_or_vassal = {
			limit = {
				NOT = {
					has_opinion_modifier = {
						name = opinion_decreased_council_power
						who = PREV
					}
				}
			}

			opinion = {
				name = opinion_decreased_council_power
				who = ROOT
				years = 5
			}
		}
	}
}

	any_de_jure_vassal = {
		limit = {
			OR = {
				religion = ROOT
				is_vassal_or_below_of = ROOT
			}
		}

		opinion = {
			name = opinion_increased_authority
			who = ROOT
			years = 5
		}
	}
}

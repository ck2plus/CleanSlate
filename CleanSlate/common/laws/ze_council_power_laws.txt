				text = council_off_custom
			}
			custom_tooltip = {
				text = tooltip_disables_advisors
			}

			hidden_effect = {
				revoke_law = law_voting_power_1

				if = {
					limit = {
						has_law = law_voting_power_1
						is_primary_holder_title = yes
					}

					holder_scope = {
						any_courtier_or_vassal = {
							limit = {
								NOT = {
									has_opinion_modifier = {
										who = ROOT

			factor = 0.05

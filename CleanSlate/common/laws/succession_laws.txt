	succ_open_elective = {
		potential = {
			OR = {
				holy_order = yes

				# Not temporary vassal revolt titles, only provincial revolts, adventurers etc.
				trigger_if = {
					limit = { temporary = yes }
					holder_scope = { NOT = { war_with = liege_before_war } }
				}

				trigger_if = {
					limit = {
						is_primary_type_title = yes
						controls_religion = yes
					}

					# Non-landless religion head titles can have normal succession laws
					# Also see event 670 in base_on_action_events.txt
					is_landless_type_title = yes
				}

				holder_scope = {
					trigger_if = {
						limit = { is_republic = yes }
						is_patrician = no
						is_merchant_republic = no
					}
					trigger_else = {
						is_theocracy = yes
					}
				}
			}

			NOR = {
				is_law_potential = succ_papal_succession
				is_law_potential = succ_catholic_bishopric
				is_law_potential = succ_offmap_succession

#
# Note: if 'chance' evaluates to 0, an AI character will also abandon the focus

focus_ed_diplomacy = {
	type = character

	modifier_name = EDUCATION_FOCUS_DIPLOMACY

	good_traits = {
		affectionate
		curious
	}

	bad_traits = {
		willful
		fussy
	}

	# Modifiers
	diplomacy = 1

	potential = {
		age >= 12
		is_adult = no
	}

	# Focuses have no success conditions
	success = {
		always = no
	}

	abort = {
		is_adult = yes
	}

	# AI Pick chance
	chance = {
		factor = 100
	}
}

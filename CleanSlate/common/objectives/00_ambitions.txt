			factor = 1.25
			from_ruler_dynasty = yes
		}
		mult_modifier = {
			factor = 1.5
			trait = poet
		}
		mult_modifier = {
			factor = 1.5
			trait = falconer
		}
		mult_modifier = {
			factor = 1.5
			trait = gregarious
		}

obj_become_paragon_of_virtue = {
	type = character

	potential = {
		NOT = { religion_group = indian_group }
		is_playable = yes
		is_landed = yes
		is_adult = yes
		is_incapable = no
		piety < 1000

		has_impious_trait_trigger = no

		OR = {
			has_education_learning_trigger = yes
			has_pious_trait_trigger = yes
		}
	}

	success = {
		piety >= 2000
	}

	effect = {
		change_learning = 1
		piety = 100

		if = {
			limit = { has_nickname = no }
			give_nickname = nick_the_holy
		}

		religion_authority = {
			modifier = paragon_of_virtue
			years = 20
		}

		religion_head = {
			opinion = {
				who = ROOT
				modifier = opinion_paragon
				years = 100
			}
		}

		any_vassal = {
			limit = {
				is_theocracy = yes
				religion = ROOT
			}
			opinion = {
				who = ROOT
				modifier = opinion_paragon
				years = 100
			}
		}

		# Achievement
		hidden_effect = {
			if = {
				limit = {
					ai = no
					is_ironman = yes
				}
				set_character_flag = achievement_become_paragon_of_virtue
			}
		}
	}

	abort = {
		OR = {
			has_impious_trait_trigger = yes
			is_incapable = yes
		}
	}

	chance = {
		factor = 50

		mult_modifier = {
			factor = 1.25
			trait = patient
		}
		mult_modifier = {
			factor = 0.01
			lower_tier_than = KING
			is_theocracy = no
		}
		mult_modifier = {
			factor = 10.0
			has_education_learning_trigger = yes
		}
		mult_modifier = {
			factor = 10.0
			trait = zealous
		}
	}
}

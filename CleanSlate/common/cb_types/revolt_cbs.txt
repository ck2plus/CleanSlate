# The CB used by "Peasant revolter" characters
peasant_revolt = {
	name = CB_NAME_PEASANT_CONQUEST
	war_name = WAR_NAME_PEASANT_CONQUEST
	sprite = 16
	truce_days = 365

	hostile_against_others = yes
	check_all_titles = yes # if permanent, setting this to true will check against all of someones titles, including vassal held titles
	press_claim = yes
	attacker_can_call_allies = no
	defender_can_call_allies = no
	can_ask_to_join_war = no
	is_independence = yes

	can_use = {
		# Attacker
		ROOT = {
			primary_title = { temporary = yes } # Revolter or adventurer trigger
			war = no
			NOT = { is_vassal_or_below_of = FROM }
		}
	}

	can_use_title = {
		real_tier = COUNT
	}

	is_valid_title = {
		holder_scope = {
			OR = {
				character = FROM
				is_vassal_or_below_of = FROM
			}
		}
	}

	on_add = {
		fire_haruspicy_event_effect = yes
		anger_china_effect = yes
	}

	on_success_title = {
		usurp_title_plus_barony_if_unlanded = {
			target = attacker
			type = revolt
		}

		attacker = {
			show_scope_change = no

			hidden_effect = {
				# Trigger automatic government validation, so actual government will depend on capital holding
				if = {
					limit = { is_nomadic = yes }
					set_government_type = feudal_government
				}
			}

			set_defacto_liege = attacker
		}

		defender = {
			show_scope_change = no

			prestige = -100
		}

		hidden_effect = {
			check_if_reconquista_finished_effect = yes

			defender = {
				landless_pope_effect = yes
			}
		}
	}

	on_fail_title = {
		attacker = {
			show_scope_change = no

			clear_wealth = yes
			imprison = defender

			hidden_effect = {
				add_character_modifier = {
					name = broken_spirit
					days = -1
				}
			}
		}

		defender = {
			show_scope_change = no

			prestige = 10

			hidden_effect = {
				opinion = {
					name = opinion_rebel_traitor
					who = attacker
					years = 100
				}
			}
		}
	}

	on_reverse_demand = {
		attacker = {
			show_scope_change = no

			clear_wealth = yes
			imprison = defender

			hidden_effect = {
				add_character_modifier = {
					name = broken_spirit
					days = -1
				}
			}
		}

		defender = {
			show_scope_change = no

			prestige = 20

			hidden_effect = {
				opinion = {
					name = opinion_rebel_traitor
					who = attacker
					years = 100
				}

				add_character_modifier = {
					name = crushed_revolt
					years = 10
				}
			}
		}
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}

	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}

	defender_ai_defeat_worth = {
		factor = 100
	}
}

# The CB used by "Heretic revolter" characters
heretic_revolt = {
	name = CB_NAME_HERETIC_UPRISING
	war_name = WAR_NAME_HERETIC_UPRISING
	sprite = 7
	truce_days = 365

	hostile_against_others = yes
	attacker_can_call_allies = no
	defender_can_call_allies = no
	can_ask_to_join_war = no
	check_de_jure_tier = DUKE

	can_use = {
		# Attacker
		ROOT = { has_character_flag = heretic_revolter }
	}

	is_valid = {
		# Attacker
		ROOT = {
			NOT = { religion = FROM }
		}
	}

	on_add = {
		fire_haruspicy_event_effect = yes
		anger_china_effect = yes
	}

	on_success = {
		attacker = {
			show_scope_change = no

			prestige = 200
			piety = 100
			occupy_minors_of_occupied_settlements = defender

			gain_all_occupied_titles = {
				who = defender
				type = revolt
			}

			clr_character_flag = heretic_revolter
		}

		defender = {
			show_scope_change = no

			prestige = -500
			piety = -500

			religion_authority = {
				name = lost_heretic_revolt
			}
		}
	}

	on_success_title = {
		attacker = {
			vassalize_or_take_under_title = {
				title = PREV
				enemy = defender
				same_religion = yes # Only vassalize rulers of my religion
				is_religious = yes
				type = holy_war
			}
		}

		if = {
			limit = {
				OR = {
					AND = {
						holder_scope = {
							OR = {
								character = defender
								is_vassal_or_below_of = defender
							}
						}

						controls_religion = no # Don't usurp religious leadership titles
					}

					has_holder = no # Create title if it doesn't exist
				}
			}

			usurp_title = {
				target = attacker
				type = holy_war
			}
		}

		hidden_effect = {
			check_if_reconquista_finished_effect = yes

			defender = {
				landless_pope_effect = yes
			}
		}
	}

	on_fail = {
		attacker = {
			show_scope_change = no

			clear_wealth = yes
			imprison = defender
			clr_character_flag = heretic_revolter

			hidden_effect = {
				add_character_modifier = {
					name = broken_spirit
					days = -1
				}
			}
		}

		defender = {
			show_scope_change = no

			prestige = 50
			piety = 100

			religion_authority = {
				name = won_heretic_revolt
			}

			hidden_effect = {
				opinion = {
					name = opinion_rebel_traitor
					who = attacker
					years = 100
				}
			}
		}
	}

	on_reverse_demand = {
		attacker = {
			show_scope_change = no

			clear_wealth = yes
			imprison = defender
			clr_character_flag = heretic_revolter

			hidden_effect = {
				add_character_modifier = {
					name = broken_spirit
					days = -1
				}
			}
		}

		defender = {
			show_scope_change = no

			prestige = 100
			piety = 200

			religion_authority = {
				name = won_heretic_revolt
			}

			hidden_effect = {
				opinion = {
					name = opinion_rebel_traitor
					who = attacker
					years = 100
				}

				add_character_modifier = {
					name = crushed_revolt
					years = 10
				}
			}
		}
	}

	attacker_ai_victory_worth = {
		factor = 100
	}

	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}

	defender_ai_defeat_worth = {
		factor = 150
	}
}

# The CB used by "Nationalist Revolter" characters
liberation_revolt = {
	name = CB_NAME_LIBERATION_REVOLT
	war_name = WAR_NAME_LIBERATION_REVOLT
	sprite = 16
	truce_days = 365

	hostile_against_others = yes
	check_de_jure_tier = KING # this scans all dejure duchies for the counties which are held by or vassals(or below) of selected character
	press_claim = yes
	can_ask_to_join_war = no
	is_independence = yes

	can_use = {
		# Attacker
		ROOT = { has_character_flag = nationalist_rebel }
	}

	can_use_title = {
		tier = KING
		has_holder = no
	}

	is_valid = {
		# Attacker
		ROOT = {
			NOT = { culture = FROM }
		}
	}

	is_valid_title = {
		has_holder = no
	}

	on_add = {
		fire_haruspicy_event_effect = yes
		anger_china_effect = yes
	}

	on_success = {
		attacker = {
			show_scope_change = no

			prestige = 1000
			occupy_minors_of_occupied_settlements = defender

			gain_all_occupied_titles = {
				who = defender
				type = revolt
			}

			clr_character_flag = nationalist_rebel
			give_nickname = nick_the_liberator
		}

		defender = {
			show_scope_change = no

			prestige = -500
		}
	}

	on_success_title = {
		custom_tooltip = {
			text = tribal_invasion_succ_tip

			attacker = {
				vassalize_or_take_under_title = {
					title = PREV
					enemy = defender
					is_revolt = yes
					type = revolt
				}

				if = { # Usurp or create kingdom title if enough land was taken
					limit = {
						num_of_count_titles >= 5

						PREV = {
							OR = {
								has_holder = no
								holder = defender
							}
						}
					}

					usurp_title = {
						target = PREV
						type = revolt
					}
				}
			}
		}

		hidden_effect = {
			check_if_reconquista_finished_effect = yes

			defender = {
				landless_pope_effect = yes
			}
		}
	}

	on_fail_title = {
		attacker = {
			show_scope_change = no

			clear_wealth = yes
			imprison = defender
			clr_character_flag = nationalist_rebel

			hidden_effect = {
				add_character_modifier = {
					name = broken_spirit
					days = -1
				}

				remove_trait = inspiring_leader
			}
		}

		defender = {
			show_scope_change = no

			prestige = 100

			hidden_effect = {
				opinion = {
					name = opinion_rebel_traitor
					who = attacker
					years = 100
				}
			}
		}
	}

	on_reverse_demand = {
		attacker = {
			show_scope_change = no

			clear_wealth = yes
			imprison = defender
			clr_character_flag = nationalist_rebel

			hidden_effect = {
				add_character_modifier = {
					name = broken_spirit
					days = -1
				}

				remove_trait = inspiring_leader
			}
		}

		defender = {
			show_scope_change = no

			prestige = 200

			hidden_effect = {
				opinion = {
					name = opinion_rebel_traitor
					who = attacker
					years = 100
				}

				add_character_modifier = {
					name = crushed_major_revolt
					years = 10
				}
			}
		}
	}

	attacker_ai_victory_worth = {
		factor = 100
	}

	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}

	defender_ai_defeat_worth = {
		factor = 100
	}
}

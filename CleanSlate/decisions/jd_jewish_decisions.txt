decisions = {

	borrow_money_radhanites = {
		ai_check_interval = 36

		only_playable = yes

		potential = {
			religion_group = jewish_group
		}

		allow = {
			prisoner = no

			custom_tooltip = {
				text = has_silk_road_trade_post_demesne_tt

				any_demesne_province = {
					trade_route = silk_road
					has_trade_post = yes
					owned_by = PREV
				}
			}

			NOR = {
				government = order_government
				has_character_modifier = borrowed_from_radhanites
			}

			top_liege = {
				NOT = { has_character_modifier = expelled_jewish }
			}
		}

		effect = {
			wealth = 200

			add_character_modifier = {
				name = borrowed_from_radhanites
				duration = -1
				inherit = yes
			}
		}

		ai_will_do = {
			factor = 1

			trigger = {
				trigger_if = {
					limit = { war = yes }
					wealth  < 200
				}
				trigger_else = {
					wealth < 0
				}
			}
		}
	}

	repay_money_radhanites = {
		ai_check_interval = 36

		only_playable = yes

		potential = {
			has_character_modifier = borrowed_from_radhanites
		}

		allow = {
			wealth >= 200
		}

		effect = {
			wealth = -200
			remove_character_modifier = borrowed_from_radhanites
		}

		ai_will_do = {
			factor = 1

			trigger = {
				wealth > 350
			}
		}
	}

}

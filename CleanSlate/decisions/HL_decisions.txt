	become_genghis_khan = {
		ai_check_interval = 60

		is_high_prio = yes

		only_independent = yes

		potential = {
			has_dlc = "Horse Lords"
			independent = yes
			culture = mongol
			is_nomadic = yes
			NOT = { has_global_flag = mongol_horde_united }
		}

		allow = {
			num_of_count_titles_in_realm >= 35
			prestige >= 1500
		}

		effect = {
			set_global_flag = mongol_horde_arrival
			set_global_flag = mongol_horde_united
			set_special_character_title = GENGHIS_KHAN
			prestige = 1000

			add_character_modifier = {
				name = greatest_of_khans
				duration = -1
			}

			if = {
				limit = {
					has_dlc = "Holy Fury"

					NOT = {
						any_owned_bloodline = {
							has_bloodline_flag = genghis_khan_line
							bloodline_is_active_for = PREV
						}
					}
				}

				create_bloodline = {
					type = genghis_khan
				}
			}

			if = {
				limit = {
					e_mongol_empire = {
						has_holder = no
					}
				}

				e_mongol_empire = {
					hidden_effect = {
						activate_title = {
							title = THIS
							status = yes
						}
					}

					grant_title = PREV
					make_primary_title = yes
				}

				any_demesne_title = {
					limit = {
						real_tier = EMPEROR
						NOT = { title = e_mongol_empire }
					}

					hidden_effect = {
						activate_title = {
							title = THIS
							status = no
						}
					}

					destroy_landed_title = THIS
				}
			}

			if = {
				limit = { ai = yes }
				character_event = { id = 60530 } # AI reinforcements
				population = 50000
			}

			narrative_event = { id = 60520 } # Notify everyone
		}

		ai_will_do = {
			factor = 1
		}
	}
